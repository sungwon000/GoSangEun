CREATE OR REPLACE TRIGGER RENICKNAME_TRI
AFTER UPDATE ON MEMBER
FOR EACH ROW
BEGIN
    UPDATE REVIEW
    SET USER_NAME = :NEW.USER_NICKNAME
    WHERE USER_ID = :NEW.USER_ID;
END;
/

-- 회원용 오더테이블에서 결제완료(Y) 된 테이블을 지울시 히스토리 테이블로 자동 INSERT
CREATE OR REPLACE TRIGGER ORDER_HISTORY_TRI
AFTER UPDATE ON SHOP_ORDER
FOR EACH ROW
WHEN (NEW.ORDER_CHECK='Y')
BEGIN
    INSERT INTO ORDER_HISTORY VALUES(
    :NEW.ORDER_NO,
    :NEW.ORDER_GROUP,
    :NEW.SHOP_ID,
    :NEW.USER_ID,
    :NEW.MENU_NO,
    :NEW.ORDER_COUNT,
    :NEW.ORDER_SUM,
    :NEW.ORDER_INFO,
    :NEW.ORDER_CHECK,SYSDATE,
    :NEW.DV_ADDRESS);
END;
/
